    def celsius_para_fahrenheit(celsius):
    """
    Converte a temperatura de Celsius para Fahrenheit.
    
    :param celsius: Temperatura em graus Celsius.
    :return: Temperatura em graus Fahrenheit.
    """
    fahrenheit = celsius * 9/5 + 32
    return fahrenheit
